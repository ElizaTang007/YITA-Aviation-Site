3

10行起飞

Python 
from stratoos import Drone

# 连接无人机
drone = Drone.connect("DRONE-001")

# 起飞
drone.takeoff(height=50)

# 移动到目标点
drone.move_to(lat=39.9042, lng=116.4074)

# 环绕飞行
drone.orbit(radius=100, circles=3)

# 降落
drone.land()

print("任务完成！")
